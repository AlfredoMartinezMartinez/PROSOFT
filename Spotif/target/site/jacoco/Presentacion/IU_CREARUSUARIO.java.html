    private void jButtonCrearUsuarioActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButtonCrearUsuarioActionPerformed
        // TODO add your handling code here:
        
        try{
                //registrado.setText("");
                String correo= jTextFieldCorreo.getText();
                String login= jTextFieldLogin.getText();
                String password= jTextFieldpass.getText();
         
         
          boolean autenticar1=GestorUsuario.crearUsuario(jTextFieldNombre.getText(),jTextFieldApellidos.getText(),login,password,correo,"usuario");
                
		if (autenticar1){
		                      
                    registrado.setText("Usuario registrado");
                    IU_LOGIN framelog=new IU_LOGIN();
                    framelog.getFrmVentanaLogIn().setVisible(true);
                    this.setVisible(false);
                    
			}
		    else{
		                                 
                         registrado.setText("Nombre Usuario de cliente ya existe"); 
                         jTextFieldNombre.setText("");
                         jTextFieldApellidos.setText("");
                         jTextFieldpass.setText("");
                         jTextFieldLogin.setText("");
                         jTextFieldCorreo.setText("");
                         
			} 
					
		}catch (Exception e){
		     e.printStackTrace();						
		    }//fin catch
    }//GEN-LAST:event_jButtonCrearUsuarioActionPerformed
